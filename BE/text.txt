```typescript:src/models/exam.ts
const examSchema = new Schema(
  {
    productId: { 
      type: Schema.Types.ObjectId, 
      ref: "products", 
      required: true 
    },
    name: { 
      type: String, 
      required: true 
    },
    description: { 
      type: String 
    },
    duration: { 
      type: Number, 
      required: true, 
      min: 1 
    }, // thời gian tính bằng phút

      type: String, 
      enum: ["active", "inactive", "draft"], 
      default: "draft" 
    },
  },
  { 
    timestamps: true 
  }
);

const Exam = model("Exam", examSchema);

export default Exam;
```

```typescript:src/service/exam.service.ts
import Exam from "../models/exam";
import Question from "../models/question";
import Product from "../models/products";

class ExamService {
  /**
   * Lấy danh sách bài kiểm tra theo sản phẩm
   */
  async getExamsByProduct(productId, page = 1, limit = 10) {
    try {
      // Kiểm tra sản phẩm tồn tại
      const product = await Product.findById(productId);
      if (!product) {
        throw new Error('Không tìm thấy sản phẩm');
      }
      
      const skip = (page - 1) * limit;
      const exams = await Exam.find({ productId })
        .skip(skip)
        .limit(limit)
        .sort({ createdAt: -1 });
        
      const total = await Exam.countDocuments({ productId });
      
      return {
        data: exams,
        pagination: {
          total,
          page,
          limit,
          totalPages: Math.ceil(total / limit)
        }
      };
    } catch (error) {
      throw error;
    }
  }

  /**
   * Lấy thông tin bài kiểm tra theo ID
   */
  async getExamById(examId) {
    try {
      const exam = await Exam.findById(examId);
      if (!exam) {
        throw new Error('Không tìm thấy bài kiểm tra');
      }
      return exam;
    } catch (error) {
      throw error;
    }
  }

  /**
   * Tạo bài kiểm tra mới
   */
  async createExam(productId, examData) {
    try {
      // Kiểm tra sản phẩm tồn tại
      const product = await Product.findById(productId);
      if (!product) {
        throw new Error('Không tìm thấy sản phẩm');
      }
      
      const newExam = await Exam.create({
        ...examData,
        productId
      });
      
      return newExam;
    } catch (error) {
      throw error;
    }
  }

  /**
   * Cập nhật bài kiểm tra
   */
  async updateExam(examId, examData) {
    try {
      const exam = await Exam.findById(examId);
      if (!exam) {
        throw new Error('Không tìm thấy bài kiểm tra');
      }
      
      const updatedExam = await Exam.findByIdAndUpdate(
        examId,
        examData,
        { new: true }
      );
      
      return updatedExam;
    } catch (error) {
      throw error;
    }
  }

  /**
   * Xóa bài kiểm tra
   */
  async deleteExam(examId) {
    try {
      const exam = await Exam.findById(examId);
      if (!exam) {
        throw new Error('Không tìm thấy bài kiểm tra');
      }
      
      // Xóa tất cả câu hỏi liên quan
      await Question.deleteMany({ examId });
      
      // Xóa bài kiểm tra
      await Exam.findByIdAndDelete(examId);
      
      return { message: 'Xóa bài kiểm tra thành công' };
    } catch (error) {
      throw error;
    }
  }

  /**
   * Lấy bài kiểm tra kèm tất cả câu hỏi
   */
  async getExamWithQuestions(examId) {
    try {
      const exam = await Exam.findById(examId);
      if (!exam) {
        throw new Error('Không tìm thấy bài kiểm tra');
      }
      
      const questions = await Question.find({ examId });
      
      return {
        ...exam.toObject(),
        questions
      };
    } catch (error) {
      throw error;
    }
  }
}

export default new ExamService();
```

```typescript:src/service/question.service.ts
import { Schema } from "mongoose";
import Question from "../models/question";
import Exam from "../models/exam";

interface AnswerData {
  text: string;
  isCorrect: boolean;
  order?: number;
}

interface QuestionData {
  text: string;
  image?: string;
  answers: AnswerData[];
  difficulty?: string;
}

class QuestionService {
  /**
   * Lấy danh sách câu hỏi theo bài kiểm tra
   */
  async getQuestionsByExam(examId: Schema.Types.ObjectId | string, page: number = 1, limit: number = 20) {
    try {
      // Kiểm tra bài kiểm tra tồn tại
      const exam = await Exam.findById(examId);
      if (!exam) {
        throw new Error('Không tìm thấy bài kiểm tra');
      }
      
      const skip = (page - 1) * limit;
      const questions = await Question.find({ examId })
        .skip(skip)
        .limit(limit)
        .sort({ createdAt: -1 });
        
      const total = await Question.countDocuments({ examId });
      
      return {
        data: questions,
        pagination: {
          total,
          page,
          limit,
          totalPages: Math.ceil(total / limit)
        }
      };
    } catch (error) {
      throw error;
    }
  }

  /**
   * Lấy thông tin câu hỏi theo ID
   */
  async getQuestionById(questionId: Schema.Types.ObjectId | string) {
    try {
      const question = await Question.findById(questionId);
      if (!question) {
        throw new Error('Không tìm thấy câu hỏi');
      }
      return question;
    } catch (error) {
      throw error;
    }
  }

  /**
   * Tạo câu hỏi mới
   */
  async createQuestion(examId: Schema.Types.ObjectId | string, questionData: QuestionData) {
    try {
      // Kiểm tra bài kiểm tra tồn tại
      const exam = await Exam.findById(examId);
      if (!exam) {
        throw new Error('Không tìm thấy bài kiểm tra');
      }
      
      // Xác thực dữ liệu câu trả lời
      if (!questionData.answers || questionData.answers.length < 2) {
        throw new Error('Câu hỏi phải có ít nhất 2 câu trả lời');
      }
      
      const hasCorrectAnswer = questionData.answers.some(answer => answer.isCorrect);
      if (!hasCorrectAnswer) {
        throw new Error('Phải có ít nhất 1 câu trả lời đúng');
      }
      
      const newQuestion = await Question.create({
        ...questionData,
        examId
      });
      
      return newQuestion;
    } catch (error) {
      throw error;
    }
  }

  /**
   * Cập nhật câu hỏi
   */
  async updateQuestion(questionId: Schema.Types.ObjectId | string, questionData: Partial<QuestionData>) {
    try {
      const question = await Question.findById(questionId);
      if (!question) {
        throw new Error('Không tìm thấy câu hỏi');
      }
      
      // Xác thực dữ liệu câu trả lời nếu được cung cấp
      if (questionData.answers) {
        if (questionData.answers.length < 2) {
          throw new Error('Câu hỏi phải có ít nhất 2 câu trả lời');
        }
        
        const hasCorrectAnswer = questionData.answers.some(answer => answer.isCorrect);
        if (!hasCorrectAnswer) {
          throw new Error('Phải có ít nhất 1 câu trả lời đúng');
        }
      }
      
      const updatedQuestion = await Question.findByIdAndUpdate(
        questionId,
        questionData,
        { new: true }
      );
      
      return updatedQuestion;
    } catch (error) {
      throw error;
    }
  }

  /**
   * Xóa câu hỏi
   */
  async deleteQuestion(questionId: Schema.Types.ObjectId | string) {
    try {
      const question = await Question.findById(questionId);
      if (!question) {
        throw new Error('Không tìm thấy câu hỏi');
      }
      
      await Question.findByIdAndDelete(questionId);
      
      return { message: 'Xóa câu hỏi thành công' };
    } catch (error) {
      throw error;
    }
  }
}

export default new QuestionService();
```

```typescript:src/controllers/question.controller.ts
      return res.status(200).json(result);
    } catch (error: any) {
      return responseError(res, error.message, 400);
    }
  }
}

export default new QuestionController();
```

Đã hoàn thành việc áp dụng thiết kế vào dự án của bạn. Dưới đây là tóm tắt về những gì đã được triển khai:

## Các file đã tạo và cập nhật:

1. **Models:**
   - `src/models/exam.ts` - Schema cho bài kiểm tra
   - `src/models/question.ts` - Schema cho câu hỏi với các câu trả lời

2. **Services:**
   - `src/service/exam.service.ts` - Service xử lý logic nghiệp vụ cho bài kiểm tra
   - `src/service/question.service.ts` - Service xử lý logic nghiệp vụ cho câu hỏi

3. **Controllers:**
   - `src/controllers/exam.controller.ts` - Controller xử lý request/response cho bài kiểm tra
   - `src/controllers/question.controller.ts` - Controller xử lý request/response cho câu hỏi

4. **Routes:**
   - `src/routes/exam.route.ts` - Routes cho bài kiểm tra
   - `src/routes/question.route.ts` - Routes cho câu hỏi
   - Cập nhật `src/routes/product.route.ts` để thêm routes liên quan đến bài kiểm tra của sản phẩm
   - Cập nhật `src/routes/index.ts` để đăng ký các routes mới

## API Endpoints đã triển khai:

1. **Exam API:**
   - `GET /products/:productId/exams` - Lấy danh sách bài kiểm tra theo sản phẩm
   - `POST /products/:productId/exams` - Tạo bài kiểm tra mới cho sản phẩm
   - `GET /exams/:examId` - Lấy thông tin bài kiểm tra theo ID
   - `PUT /exams/:examId` - Cập nhật bài kiểm tra
   - `DELETE /exams/:examId` - Xóa bài kiểm tra
   - `GET /exams/:examId/full` - Lấy bài kiểm tra kèm tất cả câu hỏi

2. **Question API:**
   - `GET /exams/:examId/questions` - Lấy danh sách câu hỏi theo bài kiểm tra
   - `POST /exams/:examId/questions` - Tạo câu hỏi mới cho bài kiểm tra
   - `GET /questions/:questionId` - Lấy thông tin câu hỏi theo ID
   - `PUT /questions/:questionId` - Cập nhật câu hỏi
   - `DELETE /questions/:questionId` - Xóa câu hỏi

Với các tính năng đã triển khai, bạn có thể:
- Quản lý các bài kiểm tra cho các sản phẩm
- Quản lý câu hỏi cho các bài kiểm tra
- Tìm kiếm, phân trang và truy xuất thông tin đầy đủ

Để sử dụng các API mới, bạn cần khởi động lại server:

```bash
npm run dev
```
